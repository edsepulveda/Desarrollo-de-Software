1# Magic utility that "redirects" to pywintypesxx.dll 

2import importlib.util, importlib.machinery, sys, os 

3def __import_pywin32_system_module__(modname, globs): 

4 # This has been through a number of iterations. The problem: how to 

5 # locate pywintypesXX.dll when it may be in a number of places, and how 

6 # to avoid ever loading it twice. This problem is compounded by the 

7 # fact that the "right" way to do this requires win32api, but this 

8 # itself requires pywintypesXX. 

9 # And the killer problem is that someone may have done 'import win32api' 

10 # before this code is called. In that case Windows will have already 

11 # loaded pywintypesXX as part of loading win32api - but by the time 

12 # we get here, we may locate a different one. This appears to work, but 

13 # then starts raising bizarre TypeErrors complaining that something 

14 # is not a pywintypes type when it clearly is! 

15 

16 # So in what we hope is the last major iteration of this, we now 

17 # rely on a _win32sysloader module, implemented in C but not relying 

18 # on pywintypesXX.dll. It then can check if the DLL we are looking for 

19 # lib is already loaded. 

20 # See if this is a debug build. 

21 suffix = '_d' if '_d.pyd' in importlib.machinery.EXTENSION_SUFFIXES else '' 

22 filename = "%s%d%d%s.dll" % \ 

23 (modname, sys.version_info[0], sys.version_info[1], suffix) 

24 if hasattr(sys, "frozen"): 

25 # If we are running from a frozen program (py2exe, McMillan, freeze) 

26 # then we try and load the DLL from our sys.path 

27 # XXX - This path may also benefit from _win32sysloader? However, 

28 # MarkH has never seen the DLL load problem with py2exe programs... 

29 for look in sys.path: 

30 # If the sys.path entry is a (presumably) .zip file, use the 

31 # directory 

32 if os.path.isfile(look): 

33 look = os.path.dirname(look) 

34 found = os.path.join(look, filename) 

35 if os.path.isfile(found): 

36 break 

37 else: 

38 raise ImportError("Module '%s' isn't in frozen sys.path %s" % (modname, sys.path)) 

39 else: 

40 # First see if it already in our process - if so, we must use that. 

41 import _win32sysloader 

42 found = _win32sysloader.GetModuleFilename(filename) 

43 if found is None: 

44 # We ask Windows to load it next. This is in an attempt to 

45 # get the exact same module loaded should pywintypes be imported 

46 # first (which is how we are here) or if, eg, win32api was imported 

47 # first thereby implicitly loading the DLL. 

48 

49 # Sadly though, it doesn't quite work - if pywintypesxx.dll 

50 # is in system32 *and* the executable's directory, on XP SP2, an 

51 # import of win32api will cause Windows to load pywintypes 

52 # from system32, where LoadLibrary for that name will 

53 # load the one in the exe's dir. 

54 # That shouldn't really matter though, so long as we only ever 

55 # get one loaded. 

56 found = _win32sysloader.LoadModule(filename) 

57 if found is None: 

58 # Windows can't find it - which although isn't relevent here, 

59 # means that we *must* be the first win32 import, as an attempt 

60 # to import win32api etc would fail when Windows attempts to 

61 # locate the DLL. 

62 # This is most likely to happen for "non-admin" installs, where 

63 # we can't put the files anywhere else on the global path. 

64 

65 # If there is a version in our Python directory, use that 

66 if os.path.isfile(os.path.join(sys.prefix, filename)): 

67 found = os.path.join(sys.prefix, filename) 

68 if found is None: 

69 # Not in the Python directory? Maybe we were installed via 

70 # easy_install... 

71 if os.path.isfile(os.path.join(os.path.dirname(__file__), filename)): 

72 found = os.path.join(os.path.dirname(__file__), filename) 

73 if found is None: 

74 # We might have been installed via PIP and without the post-install 

75 # script having been run, so they might be in the 

76 # lib/site-packages/pywin32_system32 directory. 

77 # This isn't ideal as it means, say 'python -c "import win32api"' 

78 # will not work but 'python -c "import pywintypes, win32api"' will, 

79 # but it's better than nothing... 

80 import distutils.sysconfig 

81 maybe = os.path.join(distutils.sysconfig.get_python_lib(plat_specific=1), 

82 "pywin32_system32", filename) 

83 if os.path.isfile(maybe): 

84 found = maybe 

85 if found is None: 

86 # give up in disgust. 

87 raise ImportError("No system module '%s' (%s)" % (modname, filename)) 

88 # After importing the module, sys.modules is updated to the DLL we just 

89 # loaded - which isn't what we want. So we update sys.modules to refer to 

90 # this module, and update our globals from it. 

91 old_mod = sys.modules[modname] 

92 # Load the DLL. 

93 loader = importlib.machinery.ExtensionFileLoader(modname, found) 

94 spec = importlib.machinery.ModuleSpec(name=modname, loader=loader, origin=found) 

95 mod = importlib.util.module_from_spec(spec) 

96 spec.loader.exec_module(mod) 

97 

98 # Check the sys.modules[] behaviour we describe above is true... 

99 assert sys.modules[modname] is not old_mod 

100 assert sys.modules[modname] is mod 

101 # as above - re-reset to the *old* module object then update globs. 

102 sys.modules[modname] = old_mod 

103 globs.update(mod.__dict__) 

104
